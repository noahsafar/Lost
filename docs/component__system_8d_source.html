1/**
2 * @file component_system.d
3 * @brief Entity-Component-System architecture for the game engine.
4 *
5 * This module implements an ECS (Entity-Component-System) architecture along with
6 * a Camera class for world-to-screen coordinate transformations. The ECS pattern
7 * allows for flexible game object composition where entities are containers for
8 * components, and systems operate on entities with specific component types.
9 *
10 * Components include:
11 * - TransformComponent: Position, size, and rotation
12 * - TextureComponent: Rendering information
13 * - AnimationComponent: Sprite animation state
14 * - ScriptComponent: Custom behavior scripts
15 */

31/**
32 * @brief Camera class for world-to-screen coordinate transformation.
33 *
34 * The Camera manages the viewport into the game world, handling coordinate
35 * transformations between world space and screen space. It supports following
36 * targets and provides culling optimization for off-screen objects.
37 */
38class Camera {
39 float x = 0.0f; ///< Camera X position in world coordinates
40 float y = 0.0f; ///< Camera Y position in world coordinates
41 float viewportWidth = 640.0f; ///< Screen/viewport width in pixels
42 float viewportHeight = 480.0f; ///< Screen/viewport height in pixels
43 float worldWidth = 1280.0f; ///< Total world width in pixels
44 float worldHeight = 960.0f; ///< Total world height in pixels
45
46 /**
47 * @brief Camera constructor.
48 *
49 * Initializes the camera with specified world and viewport dimensions.
50 * The camera starts centered in the world.
51 *
52 * @param worldW World width in pixels (default: 1280)
53 * @param worldH World height in pixels (default: 960)
54 * @param viewW Viewport width in pixels (default: 640)
55 * @param viewH Viewport height in pixels (default: 480)
56 */
57 this(float worldW = 1280.0f, float worldH = 960.0f, float viewW = 640.0f, float viewH = 480.0f) {
58 worldWidth = worldW;
59 worldHeight = worldH;
60 viewportWidth = viewW;
61 viewportHeight = viewH;
62
63 // Start camera in center of world
64 x = (worldWidth - viewportWidth) / 2.0f;
65 y = (worldHeight - viewportHeight) / 2.0f;
66 }
67
68 // Follow target (player)
69 void followTarget(float targetX, float targetY, float targetW, float targetH) {
70
71 float targetCenterX = targetX + targetW / 2.0f;
72 float targetCenterY = targetY + targetH / 2.0f;
73
74 x = targetCenterX - viewportWidth / 2.0f;
75 y = targetCenterY - viewportHeight / 2.0f;
76
77 // Keep camera within world bounds
78 if (x < 0) x = 0;
79 if (y < 0) y = 0;
80 if (x > worldWidth - viewportWidth) x = worldWidth - viewportWidth;
81 if (y > worldHeight - viewportHeight) y = worldHeight - viewportHeight;
82 }
83
84 // Convert world to screen
85 SDL_FRect worldToScreen(const SDL_FRect worldRect) {
86 SDL_FRect screenRect;
87 screenRect.x = worldRect.x - x;
88 screenRect.y = worldRect.y - y;
89 screenRect.w = worldRect.w;
90 screenRect.h = worldRect.h;
91 return screenRect;
92 }
93
94 bool isVisible(const SDL_FRect worldRect) {
95 SDL_FRect cameraRect;
96 cameraRect.x = x;
97 cameraRect.y = y;
98 cameraRect.w = viewportWidth;
99 cameraRect.h = viewportHeight;
100
101 return SDL_HasRectIntersectionFloat(&worldRect, &cameraRect);
102 }
103
104 // Get camera bounds as rectangle
105 SDL_FRect getCameraBounds() {
106 SDL_FRect bounds;
107 bounds.x = x;
108 bounds.y = y;
109 bounds.w = viewportWidth;
110 bounds.h = viewportHeight;
111 return bounds;
112 }
113}

246 override void OnUpdate(GameObject owner){
247 auto xfComp = owner.GetComponent!TransformComponent(ComponentType.TRANSFORM);
248 if (xfComp is null) return;
249 auto xf = &xfComp.rect;
250
251 // Rotation
252 if (rotateLeft) xfComp.rotation -= rotationSpeed * g_deltaTime;
253 if (rotateRight) xfComp.rotation += rotationSpeed * g_deltaTime;
254
255 // Keep rotation in 360 range
256 if (xfComp.rotation < 0) xfComp.rotation += 360;
257 if (xfComp.rotation >= 360) xfComp.rotation -= 360;
258
259 // Movement with deltaTime
260 if (moveLeft) xf.x -= moveSpeed * g_deltaTime;
261 if (moveRight) xf.x += moveSpeed * g_deltaTime;
262 if (moveUp) xf.y -= moveSpeed * g_deltaTime;
263 if (moveDown) xf.y += moveSpeed * g_deltaTime;
264
265 // Keep player within world bounds
266 if (xf.x < 0) xf.x = 0;
267 if (xf.x > 1280 - xf.w) xf.x = 1280 - xf.w;
268 if (xf.y < 0) xf.y = 0;
269 if (xf.y > 960 - xf.h) xf.y = 960 - xf.h;
270
271 // Shooting
272 ulong now = SDL_GetTicks();
273 bool shouldShoot = false;
274 float shootAngle = 0;
275 float shootX = 0, shootY = 0;
276
277 if (shoot && (now - lastShotTime >= shotDelay)) {
278 // Shoot in direction of ship
279 shouldShoot = true;
280 shootAngle = xfComp.rotation;
281 float cx = xf.x + xf.w/2;
282 float cy = xf.y + xf.h/2;
283 shootX = cx;
284 shootY = cy;
285 }
286 else if (mouseShoot && (now - lastShotTime >= shotDelay)) {
287 // Shoot toward mouse position
288 shouldShoot = true;
289 float cx = xf.x + xf.w/2;
290 float cy = xf.y + xf.h/2;
291 shootX = cx;
292 shootY = cy;
293
294 // Calculate world mouse position
295 float worldMouseX = mouseX;
296 float worldMouseY = mouseY;
297 if (worldCamera !is null) {
298 worldMouseX = mouseX + worldCamera.x;
299 worldMouseY = mouseY + worldCamera.y;
300 }
301
302 // Calculate angle to mouse
303 import std.math : atan2, PI;
304 float dx = worldMouseX - cx;
305 float dy = worldMouseY - cy;
306 shootAngle = atan2(dy, dx) * 180.0f / PI + 90.0f;
307
308 // Make player face mouse direction
309 xfComp.rotation = shootAngle;
310 }
311
312 if (shouldShoot) {
313 auto proj = CreateDirectionalProjectile(res, shootX, shootY, 400.0f, shootAngle); // 400 pixels per second
314 if (spawnBulletCallback !is null) spawnBulletCallback(proj);
315 res.playSound("laser");
316 lastShotTime = now;
317 }
318 }
319}

428// Handle frame timing
429class AnimationComponent : IComponent {
430 struct Frame {
431 SDL_FRect rect;
432 }
433
434 Frame[] mFrames;
435 size_t[][string] mSequences;
436 string mCurrentSequence = "";
437 size_t mCurrentFrame = 0;
438 ulong mLastFrameTime = 0;
439 ulong mFrameDelay = 200;
440
441 SDL_FRect src;
442 float prevX = -1;
443 float prevY = -1;
444
445 // Load animation from JSON
446 this(string jsonPath) {
447 auto text = cast(string) readText(jsonPath);
448 auto j = parseJSON(text);
449
450 int tileW = cast(int) j["format"]["tileWidth"].integer;
451 int tileH = cast(int) j["format"]["tileHeight"].integer;
452 int sheetW = cast(int) j["format"]["width"].integer;
453 int cols = sheetW / tileW;
454
455 auto framesObj = j["frames"].object;
456
457 // Build all frames
458 Frame[] framesSet;
459 foreach(seqName, seqData; framesObj) {
460 auto indicesArray = seqData.array;
461 size_t[] frameIndices;
462 foreach(idx; indicesArray) {
463 size_t frameNum = cast(size_t) idx.integer;
464 int row = cast(int)(frameNum / cols);
465 int col = cast(int)(frameNum % cols);
466
467 SDL_FRect f;
468 f.x = col * tileW;
469 f.y = row * tileH;
470 f.w = tileW;
471 f.h = tileH;
472 framesSet ~= Frame(f);
473
474 frameIndices ~= frameNum;
475 }
476 mSequences[seqName] = frameIndices;
477 }
478
479 mFrames = framesSet;
480 if (mFrames.length > 0) {
481 src = mFrames[0].rect;
482 mCurrentSequence = mSequences.keys.length > 0 ? mSequences.keys[0] : "";
483 }
484 }
485
486 void LoopAnimationSequence(string seqName) {
487 if (!(seqName in mSequences)) return;
488
489 auto xf = this.prevX;
490 auto indices = mSequences[seqName];
491 ulong now = SDL_GetTicks();
492
493 if (now - mLastFrameTime >= mFrameDelay) {
494 mCurrentFrame = (mCurrentFrame + 1) % indices.length;
495 src = mFrames[indices[mCurrentFrame]].rect;
496 mLastFrameTime = now;
497 }
498 }
499
500 override void Update(GameObject owner) {
501 auto xf = owner.TransformRect();
502 if (xf is null) return;
503
504 bool isMoving = (xf.x != prevX) || (xf.y != prevY);
505 prevX = xf.x;
506 prevY = xf.y;
507
508 if (isMoving) {
509 // Freeze animation on first frame when moving
510 mCurrentFrame = 0;
511 if (mCurrentSequence in mSequences) {
512 auto indices = mSequences[mCurrentSequence];
513 if (indices.length > 0)
514 src = mFrames[indices[0]].rect;
515 }
516 } else {
517 // Idle animation
518 if (mCurrentSequence in mSequences) {
519 auto indices = mSequences[mCurrentSequence];
520 if (indices.length > 0) {
521 ulong now = SDL_GetTicks();
522 if (now - mLastFrameTime >= mFrameDelay) {
523 mCurrentFrame = (mCurrentFrame + 1) % indices.length;
524 src = mFrames[indices[mCurrentFrame]].rect;
525 mLastFrameTime = now;
526 }
527 }
528 }
529 }
530 }
