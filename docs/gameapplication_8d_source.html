244 void RunLoop() {
245 ulong frames = 0;
246 ulong fps = 0;
247 ulong fpsTimer = SDL_GetTicks();
248 lastFrameTime = SDL_GetTicks();
249
250 while (mGameIsRunning) {
251 ulong frameStart = SDL_GetTicks();
252
253 deltaTime = (frameStart - lastFrameTime) / 1000.0f;
254 lastFrameTime = frameStart;
255
256 if (deltaTime > 1.0f / 30.0f) {
257 deltaTime = 1.0f / 60.0f;
258 }
259
260 AdvanceFrame();
261 frames++;
262
263 if (SDL_GetTicks() - fpsTimer >= 1000) {
264 fps = frames;
265 frames = 0;
266 fpsTimer = SDL_GetTicks();
267
268 string title = "Lost Bunny - FPS: " ~ to!string(fps);
269 SDL_SetWindowTitle(mWindow, title.toStringz);
270 }
271
272 ulong frameTime = SDL_GetTicks() - frameStart;
273 if (frameTime < FRAME_DELAY) {
274 SDL_Delay(cast(uint)(FRAME_DELAY - frameTime));
275 }
276 }
