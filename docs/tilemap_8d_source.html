26/**
27 * @brief 2D tilemap for level rendering and collision.

29 * The TileMap class manages a 2D grid of tiles that form the game level.
30 * It provides functionality for loading from JSON, setting/getting tiles,

33class TileMap {
34 private int[][] grid; ///< 2D grid storing tile indices
35 private int mapWidth; ///< Width of the map in tiles
36 private int mapHeight; ///< Height of the map in tiles
37 private int tileSize; ///< Size of each tile in pixels
38 private TileSet tileset; ///< TileSet used for rendering

40 /**
41 * @brief Construct a new TileMap.
42 *
43 * Creates an empty tilemap with the specified dimensions. All tiles are
44 * initialized to -1 (empty).
45 *
46 * @param mapWidth Width of the map in tiles
47 * @param mapHeight Height of the map in tiles
48 * @param tileSize Size of each tile in pixels
49 * @param tileset TileSet to use for rendering tiles
50 */
51 this(int mapWidth, int mapHeight, int tileSize, TileSet tileset) {
52 this.mapWidth = mapWidth;
53 this.mapHeight = mapHeight;
54 this.tileSize = tileSize;
55 this.tileset = tileset;
56
57 // Initialize grid with -1 (= empty)
58 grid = new int[][](mapHeight, mapWidth);
59 for(int y = 0; y < mapHeight; y++) {
60 for(int x = 0; x < mapWidth; x++) {
61 grid[y][x] = -1;
62 }
63 }
64 }
65
66 void setTile(int mapX, int mapY, int tileIndex) {
67 if(mapX >= 0 && mapX < mapWidth && mapY >= 0 && mapY < mapHeight) {
68 grid[mapY][mapX] = tileIndex;
69 }
70 }
71
72 int getTile(int mapX, int mapY) {
73 if(mapX >= 0 && mapX < mapWidth && mapY >= 0 && mapY < mapHeight) {
74 return grid[mapY][mapX];
75 }
76 return -1;
77 }
78
79 void render(int offsetX, int offsetY) {
80 for(int y = 0; y < mapHeight; y++) {
81 for(int x = 0; x < mapWidth; x++) {
82 int tileIndex = grid[y][x];
83 if(tileIndex >= 0) { // Render if tile
84 int screenX = offsetX + x * tileSize;
85 int screenY = offsetY + y * tileSize;
86 tileset.renderTile(tileIndex, screenX, screenY);
87 }
88 }
89 }
90 }

92 bool screenToMap(int screenX, int screenY, int offsetX, int offsetY, out int mapX, out int mapY) {
93 mapX = (screenX - offsetX) / tileSize;
94 mapY = (screenY - offsetY) / tileSize;
95
96 return (mapX >= 0 && mapX < mapWidth && mapY >= 0 && mapY < mapHeight);

104 // Load tilemap from JSON
105 static TileMap loadFromJSON(string filename, TileSet tileset) {
106 import std.file : readText, exists;
107 import std.json : parseJSON;
108 import std.conv : to;
109
110 if (!exists(filename)) {
111 writeln("Error: Tilemap file not found: ", filename);
112 return null;
113 }
114
115 auto jsonData = readText(filename).parseJSON();
116
117 int width = cast(int)jsonData["width"].integer;
118 int height = cast(int)jsonData["height"].integer;
119 int tileSize = cast(int)jsonData["tileSize"].integer;
120
121 auto tilemap = new TileMap(width, height, tileSize, tileset);
122
123 auto dataArray = jsonData["data"].array;
124 for(int y = 0; y < height && y < dataArray.length; y++) {
125 auto row = dataArray[y].array;
126 for(int x = 0; x < width && x < row.length; x++) {
127 tilemap.setTile(x, y, cast(int)row[x].integer);
128 }
129 }
130
131 writeln("Loaded tilemap from: ", filename);
132 return tilemap;
133 }
