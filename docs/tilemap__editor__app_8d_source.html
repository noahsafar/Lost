294 private void drawTiles(SDL_Renderer* renderer) {
295 SDL_Event event;
296 for (int y = 0; y < gridHeight; y++) {
297 for (int x = 0; x < gridWidth; x++) {

300 SDL_FRect destRect;

302 destRect.y = y * tileSize;
303 destRect.w = tileSize;
304 destRect.h = tileSize;
305
306 // current location of the mouse overrides the tile display
307 if ((x == cast(int)(mouseX / tileSize)) && (y == cast(int)(mouseY / tileSize))) {

312
313 if (tileType == TileType.TILE && tileTexture !is null) {
314 SDL_RenderTexture(renderer, tileTexture, null, &destRect);

316 else if (tileType == TileType.SPIKE_UP && spikeTexture !is null) {

324 }
325 else if (tileType == TileType.SPIKE_LEFT && spikeTexture !is null) {
326 SDL_RenderTextureRotated(renderer, spikeTexture, null, &destRect, 270.0f, null, SDL_FLIP_NONE);

328 else if (tileType == TileType.CARROT && carrotTexture !is null) {
329 SDL_RenderTexture(renderer, carrotTexture, null, &destRect);

331 else if (tileType == TileType.BONE && carrotTexture !is null) {
332 SDL_RenderTexture(renderer, boneTexture, null, &destRect);

337 else {
338 ubyte[3] color;

341 else if (tileType == TileType.CARROT) color = [255, 150, 50];

343 SDL_SetRenderDrawColor(renderer, color[0], color[1], color[2], 255);
